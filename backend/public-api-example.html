API Usage Example:

// JavaScript/Fetch Example
const analyzeClothing = async (base64Image) => {
    const response = await fetch('http://localhost:3000/api/public/analyze-clothing', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            image: base64Image
        })
    });

    const result = await response.json();
    return result;
};

// cURL Example
curl -X POST http://localhost:3000/api/public/analyze-clothing \
  -H "Content-Type: application/json" \
  -d '{"image":"data:image/jpeg;base64,..."}'
